  const state = { mode:"nav", quizTopic:null, quizIndex:0, awaitingAnswer:false };
  const $ = (id) => document.getElementById(id);

  function norm(s){ return (s||"").toLowerCase().replace(/[^\p{L}\p{N}\s\.\-]/gu," ").replace(/\s+/g," ").trim(); }

  function audioListForSlideId(id){
    const it = CONTENT.find(x => x.id === id);
    const a = (it && it.audio) ? it.audio : [];
    if (!a.length) return "Audio: (none on this slide)";
    return "Audio:\n" + a.map((u,i)=>`${i+1}) ${u}`).join("\n");
  }

  function findMatches(query, limit=3){
    const q = norm(query);
    const scored = CONTENT.map(it=>{
      const hay = norm(it.en+" "+it.overview+" "+(it.tags||[]).join(" ")+" "+it.id);
      let s=0;
      if (q===norm(it.id)) s+=999;
      if (hay.includes(q)) s+=30;
      for (const tok of q.split(" ")) if (tok && hay.includes(tok)) s+=5;
      return {it,s};
    }).filter(x=>x.s>0).sort((a,b)=>b.s-a.s).slice(0,limit).map(x=>x.it);
    return scored;
  }

  function overviewTOC(){
    return "📚 Overview (slides)\n\n" + CONTENT.map(x=>`• Slide ${x.id}: ${x.en}`).join("\n") +
           "\n\nTip: ask “find alphabet” or “listen food”.";
  }

  function formatWhere(item){
    const aud = item.audio && item.audio.length
      ? "\n\n" + ("🔊 " + audioListForSlideId(item.id))
      : "\n\n🔇 Audio: (none on this slide)";
    return `📍 Slide ${item.id}\n**${item.en}**\nOverview: ${item.overview}${aud}`;
  }

  function addBubble(role, html){
    const wrap = $("a1-chat");
    const row = document.createElement("div");
    row.style.display="flex"; row.style.margin="8px 0";
    row.style.justifyContent = role==="user" ? "flex-end" : "flex-start";
    const bubble = document.createElement("div");
    bubble.style.maxWidth="80%"; bubble.style.padding="10px 12px";
    bubble.style.borderRadius="12px"; bubble.style.border="1px solid #e6e6e6";
    bubble.style.background = role==="user" ? "#fff" : "#f3f3f3";
    bubble.style.whiteSpace="pre-wrap";
    bubble.innerHTML = html;
    row.appendChild(bubble); wrap.appendChild(row);
    wrap.scrollTop = wrap.scrollHeight;
  }

  function lessonKey(text){
    const q = norm(text);
    if (q.includes("alphabet")) return "alphabet";
    if (q.includes("number")) return "numbers";
    if (q.includes("greet") || q.includes("hello") || q.includes("introduc") || q.includes("name")) return "greetings";
    if (q.includes("w-question") || q.includes("woher") || q.includes("wie")) return "wquestions";
    if (q.includes("pronoun")) return "pronouns";
    if (q.includes("negation") || q.includes("kein") || q.includes("nicht")) return "negation";
    if (q.includes("food") || q.includes("mensa") || q.includes("essen") || q.includes("möchte") || q.includes("wollen") || q.includes("können")) return "food";
    return null;
  }

  function setMode(m){
    state.mode=m; state.quizTopic=null; state.quizIndex=0; state.awaitingAnswer=false;
    return `✅ Mode set to **${m}**.\nTry: ` +
      (m==="nav" ? `"find food" / "overview"` :
       m==="teach" ? `"practice greetings" / "practice negation"` :
       m==="quiz" ? `"quiz numbers" / "quiz food"` :
       `"listen alphabet" / "listen food"`);
  }

  function practice(topic){
    const key = lessonKey(topic);
    if (!key || !LESSONS[key]) return "Try: practice alphabet | numbers | greetings | w-questions | pronouns | negation | food";
    const L = LESSONS[key];
    const slideId = L.key;
    const aud = "🔊 " + audioListForSlideId(slideId);
    return `📘 **${L.title}**\n📍 ${L.where}\n\n${aud}\n\n` +
      `What to know:\n• ${L.teach.join("\n• ")}\n\n`+
      `Examples:\n• ${L.examples.join("\n• ")}\n\nTry: "listen ${key}" or "quiz ${key}"`;
  }

  function listen(topic){
    const key = lessonKey(topic);
    if (!key || !LESSONS[key]) return "Try: listen alphabet | listen numbers | listen greetings | listen food";
    const L = LESSONS[key];
    const slideId = L.key;
    const aud = "🔊 " + audioListForSlideId(slideId);
    return `🎧 **Listening Trainer: ${L.title}**\n📍 ${L.where}\n\n${aud}\n\n` +
      `Steps:\n1) Listen once (no reading)\n2) Listen again + repeat aloud\n3) Pause → write what you hear (dictation)\n4) Check with slide text\n\n` +
      `Bonus: record yourself (phone voice memo) and compare pronunciation.`;
  }

  function startQuiz(topic){
    const key = lessonKey(topic) || norm(topic);
    if (!QUIZZES[key]) return "Try: quiz numbers | quiz greetings | quiz w-questions | quiz pronouns | quiz negation | quiz food";
    state.quizTopic = key; state.quizIndex=0; state.awaitingAnswer=true;
    return `📝 Quiz started: **${key}**\n\nQ1: ${QUIZZES[key][0].q}\n\nType your answer.`;
  }

  function checkQuizAnswer(userText){
    const key = state.quizTopic;
    const quiz = QUIZZES[key];
    const item = quiz[state.quizIndex];
    const u = norm(userText);
    const ok = item.a.some(ans => norm(ans)===u || u.includes(norm(ans)));
    let out = ok ? "✅ Correct!" : `❌ Not quite.\nAccepted: ${item.a.join(" / ")}`;
    state.quizIndex++;
    if (state.quizIndex >= quiz.length){
      state.awaitingAnswer=false;
      out += `\n\n🎉 Quiz finished.\nTry: "practice ${key}" or "listen ${key}".`;
      return out;
    }
    out += `\n\nQ${state.quizIndex+1}: ${quiz[state.quizIndex].q}`;
    return out;
  }

  function handleNav(query){
    const q = norm(query).replace(/^where is\s+/,"").replace(/^find\s+/,"").replace(/^which slide is\s+/,"").trim();
    if (q==="overview" || q==="toc") return overviewTOC();
    const matches = findMatches(q, 3);
    if (!matches.length) return "Not found. Try: overview, alphabet, numbers, greetings, w-questions, pronouns, negation, food.";
    let out = formatWhere(matches[0]);
    if (matches.length>1) out += `\n\nOther close matches:\n` + matches.slice(1).map(m=>`• Slide ${m.id}: ${m.en}`).join("\n");
    return out;
  }

  function reply(text){
    const q = norm(text);

    if (q==="reset"){
      state.mode="nav"; state.quizTopic=null; state.quizIndex=0; state.awaitingAnswer=false;
      return "✅ Reset done. Mode = nav. Type 'overview' or 'find numbers'.";
    }

    if (state.awaitingAnswer && !q.startsWith("mode ") && !q.startsWith("quiz ") && q!=="overview"){
      return checkQuizAnswer(text);
    }

    if (q.startsWith("mode ")){
      const m = q.replace("mode ","").trim();
      if (["nav","teach","quiz","listen"].includes(m)) return setMode(m);
      return "Modes: nav | teach | quiz | listen";
    }

    if (q==="overview" || q==="toc") return overviewTOC();

    if (q.startsWith("quiz ")){ state.mode="quiz"; return startQuiz(q.replace("quiz ","")); }
    if (q.startsWith("practice ")){ state.mode="teach"; return practice(q.replace("practice ","")); }
    if (q.startsWith("listen ")){ state.mode="listen"; return listen(q.replace("listen ","")); }

    if (state.mode==="teach") return practice(q);
    if (state.mode==="listen") return listen(q);
    if (state.mode==="quiz") return startQuiz(q);

    return handleNav(q);
  }

  function send(){
    const input = $("a1-input");
    const text = input.value.trim();
    if (!text) return;
    addBubble("user", text);
    const ans = reply(text);
    setTimeout(()=>addBubble("bot", ans), 120);
    input.value=""; input.focus();
  }

  $("a1-send").addEventListener("click", send);
  $("a1-input").addEventListener("keydown", (e)=>{ if(e.key==="Enter") send(); });

  addBubble("bot",
`Hi Masub 👋 I’m your **offline A1 assistant** with **audio links**.

Try:
• overview
• find alphabet
• listen greetings
• practice negation
• quiz food
Or: mode teach / mode quiz / mode listen`
  );
})();
</script>
